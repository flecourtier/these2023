	\only<2>{\textbf{Parametric Poisson problem (with homogeneous Dirichlet BC):} \\
	For one or several  $\bm{\mu}\in \mathcal{M}$, find $u: \Omega\to \mathbb{R}$ such that
	\begin{equation}\label{eq:ob_pde}\tag{$\mathcal{P}$}
		\left\{
		\begin{aligned}
			-\Delta u(\bm{x};\bm{\mu}) & = f(\bm{x};\bm{\mu}), \; &  & (\bm{x},\bm{\mu}) \in \Omega \times \mathcal{M}, \\
			u(\bm{x};\bm{\mu}) & =0, \;  &  & (\bm{x},\bm{\mu}) \in \partial \Omega \times \mathcal{M},
		\end{aligned}
		\right.
	\end{equation}
	with $\Omega \subset \mathbb{R}^d$ a domain ($d=1,2,3$) and $\mathcal{M} \subset \mathbb{R}^p$ the parameter space ($p$ the number of parameters).
