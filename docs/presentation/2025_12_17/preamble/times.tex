\newcommand{\PreprocessSum}[2]{%
    \pgfplotstableread[col sep=comma]{#1}{#2}%
    \pgfplotstablecreatecol[
        create col/expr={\thisrow{err_U_1}+\thisrow{err_U_2}+\thisrow{err_T}} %+\thisrow{err_p}
    ]{err_sum}{#2}%
}
